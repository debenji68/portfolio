Générateur de mots de passe sécurisés

Python · Tkinter · module random · NSI — Lycée Théodore Deck

Application Python avec interface graphique Tkinter développée en classe de Terminale NSI. Elle génère un mot de passe aléatoire entre 6 et 18 caractères en combinant lettres, chiffres et symboles. L'interface propose un menu Fichier (Nouveau, Quitter) et un bouton central de génération. Essayez la version interactive ci-dessous !

import tkinter
import string
import random

#variable globale
lgFen=720; hFen=480
couleurDeFonds='#406599'

def generer_mdp():
    """Génération de mot de passe aléatoire"""
    mdp_mini=6; mdp_maxi=18
    caracteres_autorises = string.ascii_letters + string.punctuation + string.digits
    mdp = "".join(
        random.choice(caracteres_autorises)
        for x in range(random.randint(mdp_mini, mdp_maxi))
    )
    champ_mdp.delete(0, 'end')
    champ_mdp.insert(0, mdp)
    return None

# Création de la fenêtre
window = tkinter.Tk()
window.title(" Générateur de mot de passe")
window.geometry(f"{lgFen}x{hFen}")
window.config(background=couleurDeFonds)

# Frame principale + image + frame droite
frame = tkinter.Frame(window, bg=couleurDeFonds)
frame.pack(expand='yes')
img = tkinter.PhotoImage(file='login2.png').zoom(1).subsample(2)
canvas = tkinter.Canvas(frame, width=lgFen//2, height=hFen,
    bg=couleurDeFonds, bd=0, highlightthickness=0)
canvas.create_image(lgFen//4, hFen//2, image=img)
canvas.grid(row=0, column=0)

frame_droite = tkinter.Frame(frame, bg=couleurDeFonds)
frame_droite.grid(row=0, column=1)

# Widgets
titre = tkinter.Label(frame_droite, text="Mot de passe",
    font=("Courrier",20), bg=couleurDeFonds, fg='#FFFFFF')
titre.pack()

champ_mdp = tkinter.Entry(frame_droite, font=("Courier",20),
    bg=couleurDeFonds, fg='#FFFFFF')
champ_mdp.pack()

BP_mdp = tkinter.Button(frame_droite, text="Générer le mot de passe",
    font=("Courrier",20), bg=couleurDeFonds, fg='#FFFFFF',
    command=generer_mdp)
BP_mdp.pack(fill=tkinter.BOTH, pady=20)

# Barre de menu Fichier
menu_barre = tkinter.Menu(window)
menu_fichier = tkinter.Menu(menu_barre, tearoff=0)
menu_fichier.add_command(label="Nouveau", command=generer_mdp)
menu_fichier.add_command(label="Quitter", command=window.destroy)
menu_barre.add_cascade(label="Fichier", menu=menu_fichier)
window.config(menu=menu_barre)

window.mainloop()
